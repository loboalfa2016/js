async function getCoderData(codeurl) {
    try {
        console.log("Intentando conectar a:", codeurl);
        const response = await fetch(codeurl);
        
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        
        const coders = await response.json();
        console.log("Datos recibidos:", coders);
        
        // Mostrar coders visualmente
        displayCoders(coders);
        
        // Pedir nombre al usuario
        console.log("A punto de mostrar prompt...");
        const userName = prompt("¿Cuál es tu nombre?");
        console.log("Nombre ingresado:", userName);
        console.log("Tipo de userName:", typeof userName);
        
        if (!userName) {
            console.log("No se ingresó ningún nombre");
            return;
        }
        
        console.log("Buscando en", coders.length, "coders...");
        console.log("Nombres disponibles:", coders.map(c => c.name));
        
        // Buscar coder sin importar mayúsculas/minúsculas
        const foundCoder = coders.find(coder => {
            console.log(`Comparando "${coder.name}" con "${userName}"`);
            return coder.name.toLowerCase() === userName.toLowerCase();
        });
        console.log("Coder encontrado:", foundCoder);
        
        if (foundCoder) {
            // Validación especial para Emanuel
            if (foundCoder.name.toLowerCase() === "emanuel") {
                const email = prompt("Ingresa tu email:");
                const password = prompt("Ingresa tu contraseña:");
                
                if (email === foundCoder.email && password === foundCoder.password) {
                    alert("¡Autenticación exitosa! Bienvenido Emanuel");
                } else {
                    alert("Email o contraseña incorrectos");
                }
            } else {
                // Para otros coders, mostrar saludo normal
                alert(`¡Hola ${foundCoder.name}!`);
            }
        } else {
            alert("No hay nombres que coincidan con tu búsqueda");
        }
    } catch (error) {
        console.error("Error en la petición:", error);
        alert("Error al conectar con el servidor. Asegúrate de que el servidor JSON está corriendo en localhost:3000");
    }
}

// Función para mostrar coders en el HTML
function displayCoders(coders) {
    const codersList = getElement("coders-list");
    if (codersList) {
        codersList.innerHTML = "";
        
        coders.forEach(coder => {
            const coderCard = document.createElement("div");
            coderCard.className = "bg-white p-4 rounded-lg shadow-md border";
            coderCard.innerHTML = `
                <h3 class="font-bold text-lg">${coder.name}</h3>
                <p class="text-gray-600">Edad: ${coder.age}</p>
                <p class="text-gray-600">Status: ${coder.status ? 'Activo' : 'Inactivo'}</p>
                ${coder.email ? `<p class="text-gray-600">Email: ${coder.email}</p>` : ''}
            `;
            codersList.appendChild(coderCard);
        });
    }
}
